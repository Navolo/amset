# This is an example AMSET settings file.

# This controls the number of “densification” points added near the Fermi # level num_extra_kpoints: 5000

# The materials section holds all materials parameters required to calculate # the scattering rates material:

# dielectric constants high_frequency_dielectric: 3 static_dielectric: 6

# ACD parameters elastic_constant: 150 # in GPa deformation_potential: (9, 8) # in eV, given for the VBM and CBM deformation_potential_cbm: 9 # in eV

# PIE parameters piezeoelectric_coefficient: 0.15 # in ?

# IMP parameters acceptor_charge: 1 donor_charge: 1

# POP parameters pop_frequency: 10 # in THz
